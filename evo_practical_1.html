# 1. IMPORTS AND PARAMETERS
# Libraries
import numpy as np
import matplotlib.pyplot as plt
from matplotlib.widgets import Slider

# Parameters for simulation
GRID_SIZE = 100  # Resolution of the grid
MAX_VELOCITY = 0.02  # Maximum velocity magnitude
MAX_FORCE = 0.02  # Maximum force magnitude
BOID_SPEED = 1  # SPEED OF BOIDS, CAN BE VARIABLE
TARGET_REACHED_THRESHOLD = 0.05  # Distance threshold to consider the target reached
DRAG_COEFFICIENT = 0.1  # Friction to slow down the cell naturally

# 2. MAIN LOOP (using functions and classes defined below)
def main():
    """Main function to set up and run the simulation."""
    # Initialise simulation
    num_cells = 100
    sim = Simulation(num_cells)

    # Enable interactive mode
    plt.ion()
    fig, ax = plt.subplots(figsize=(6, 6))
    plt.subplots_adjust(bottom=0.2)

    # Add a slider for selecting the number of cells
    ax_slider = plt.axes([0.2, 0.05, 0.6, 0.03])
    slider = Slider(ax_slider, 'Cells', 1, 50, valinit=num_cells, valstep=1)

    def update_cells(val):
        sim.initialise_cells(int(slider.val))

    slider.on_changed(update_cells)

    # Run simulation
    for t in range(1, 10000):
        sim.simulate_step()

        # Clear the plot
        ax.clear()

        # Plot the Gaussian grid as a background
        ax.imshow(sim.grid, extent=(0, 1, 0, 1), origin='lower', cmap='viridis', alpha=0.5)

        # Plot the target as an orange dot
        ax.scatter(sim.target_position[0], sim.target_position[1], c='orange', s=50, edgecolor='white')

        # Plot each cell as a blue dot and add a quiver to show the velocity direction
        for cell in sim.cells:
            ax.scatter(cell.x, cell.y, c='blue', s=100, edgecolor='black')
            ax.quiver(cell.x, cell.y, cell.vx * 0.35, cell.vy * 0.35, angles='xy', scale_units='xy', scale=0.05, color='darkblue')

        # Set plot limits and labels
        ax.set_xlim(0, 1)
        ax.set_ylim(0, 1)
        ax.set_aspect('equal', adjustable='box')
        ax.set_title(f"Timestep: {t}")
        ax.set_xlabel("X")
        ax.set_ylabel("Y")

        plt.pause(0.01)

    # Keep the final plot open
    plt.ioff()
    plt.show()

# 3. SIMULATION CLASS
class Simulation:
    """Manages the grid, cells, target, and simulation logic."""
    def __init__(self, num_cells):
        self.grid = np.zeros((GRID_SIZE, GRID_SIZE))  # Initialise an empty grid
        self.cells = []
        self.target_position = [np.random.uniform(0, 1), np.random.uniform(0, 1)]
        self.create_gaussian(self.grid, 0.5, 0.5, 0.1)  # Create Gaussian grid
        self.initialise_cells(num_cells)

    def simulate_step(self):
        """Simulate one timestep of the simulation."""
        for cell in self.cells:
            # Reset acceleration at the start of each step
            cell.ax = 0
            cell.ay = 0

            # Apply steering behaviours
            #self.steer_towards_dot(cell)  
            
            self.avoid_collision(cell)
            self.stick_to_close(cell)
            self.find_peak(cell)
            self.check_target_reached(cell)

            # Apply drag force to acceleration
            cell.ax += -DRAG_COEFFICIENT * cell.vx
            cell.ay += -DRAG_COEFFICIENT * cell.vy

            # Apply forces and update position
            cell.apply_forces()
            cell.update_position()

            # Limit velocity to the maximum allowed
            cell.vx = np.clip(cell.vx, -MAX_VELOCITY, MAX_VELOCITY)
            cell.vy = np.clip(cell.vy, -MAX_VELOCITY, MAX_VELOCITY)

    def initialise_cells(self, num_cells):
        """Initialise the cells with random positions and velocities."""
        self.cells = []
        for _ in range(num_cells):
            x = np.random.uniform(0, 1)
            y = np.random.uniform(0, 1)
            vx = 0
            vy = 0
            speed = 0.1 + 0.9 * np.random.uniform(0, 1) * BOID_SPEED
            speed = 1
            self.cells.append(Cell(x, y, vx, vy, speed))
        self.target_position = [np.random.uniform(0, 1), np.random.uniform(0, 1)]

    def create_gaussian(self, grid, mean_x, mean_y, std_dev):
        """Create a Gaussian distribution on the grid."""
        for i in range(GRID_SIZE):
            for j in range(GRID_SIZE):
                x = i / (GRID_SIZE - 1)
                y = j / (GRID_SIZE - 1)
                distance_squared = (x - mean_x)**2 + (y - mean_y)**2
                grid[i, j] = np.exp(-distance_squared / (2 * std_dev**2)) #* np.random.uniform(0.5, 1.5)
                grid[i, j] = np.exp(-distance_squared / (2 * std_dev**2)) * np.random.uniform(0.0, 2.0)

        # Normalize the grid to keep the total resource concentration the same
        grid /= np.sum(grid)
    
    def find_peak(self, cell):
        """Make the cell move towards the peak of the resource gradient with a random walk."""
        # Convert cell position to grid indices
        grid_x = int(cell.x * (GRID_SIZE - 1))
        grid_y = int(cell.y * (GRID_SIZE - 1))

        # Ensure indices are within bounds
        grid_x = np.clip(grid_x, 1, GRID_SIZE - 2)
        grid_y = np.clip(grid_y, 1, GRID_SIZE - 2)

        # Calculate the gradient of the grid at the cell's position
        grad_x = self.grid[grid_x + 1, grid_y] - self.grid[grid_x - 1, grid_y]
        grad_y = self.grid[grid_x, grid_y + 1] - self.grid[grid_x, grid_y - 1]

        # Normalize the gradient to get the direction
        grad_magnitude = np.sqrt(grad_x**2 + grad_y**2)
        if grad_magnitude > 0:
            grad_x /= grad_magnitude
            grad_y /= grad_magnitude

        # Check if the cell is moving in the right direction
        dot_product = grad_x * cell.vx + grad_y * cell.vy  # Dot product of gradient and velocity
        if dot_product > 0:
            # Moving in the right direction: small random adjustment
            angle = np.random.uniform(-np.pi / 8, np.pi / 8)  # Small angle change
        else:
            # Moving in the wrong direction: large random adjustment
            angle = np.random.uniform(-np.pi / 2, np.pi / 2)  # Large angle change

        # Rotate the velocity vector by the random angle
        cos_angle = np.cos(angle)
        sin_angle = np.sin(angle)
        new_vx = cell.vx * cos_angle - cell.vy * sin_angle
        new_vy = cell.vx * sin_angle + cell.vy * cos_angle

        # Update the velocity with the new direction
        cell.vx = new_vx
        cell.vy = new_vy

        # Apply a small force in the direction of the gradient
        cell.ax += grad_x * cell.speed / 1000
        cell.ay += grad_y * cell.speed / 1000
        
    
    def avoid_collision(self, cell):
        """Calculate avoidance forces to prevent cells from colliding."""
        for other_cell in self.cells:
            if other_cell is not cell:
                # Calculate the distance between the two cells
                dx = cell.x - other_cell.x
                dy = cell.y - other_cell.y
                distance = np.sqrt(dx**2 + dy**2)

                # If the cells are too close, apply a repulsion force
                if distance < 0.035 and distance > 0:  # Threshold for "too close"
                    # Calculate the repulsion force proportional to the inverse of the distance
                    force_magnitude = (0.035 - distance) / distance
                    cell.x += force_magnitude * dx
                    cell.y += force_magnitude * dy
                    
    def stick_to_close(self, cell):
        """Stick to closeby cells."""
        for other_cell in self.cells:
            if other_cell is not cell:
                # Calculate the distance between the two cells
                dx = cell.x - other_cell.x
                dy = cell.y - other_cell.y
                distance = np.sqrt(dx**2 + dy**2)

                # If the cells are too close, apply a repulsion force
                if distance < 0.05 and distance > 0:  # Threshold for "close"
                    # Calculate the repulsion force proportional to the inverse of the distance
                    force_magnitude = 0.35*(0.05 - distance) / distance
                    cell.x -= force_magnitude * dx
                    cell.y -= force_magnitude * dy                    
    
    def steer_towards_dot(self, cell):
        """Steer the cell towards the target position."""
        # Calculate the direction vector to the target
        dx = self.target_position[0] - cell.x
        dy = self.target_position[1] - cell.y
        dist_to_target = np.sqrt(dx**2 + dy**2)

        if dist_to_target > 0:
            # Normalise the direction vector
            norm_dx = dx / dist_to_target
            norm_dy = dy / dist_to_target

            # Scale the acceleration by the cell's speed
            cell.ax += norm_dx * cell.speed / 1000
            cell.ay += norm_dy * cell.speed / 1000
    
    def check_target_reached(self, cell):
        distance_to_target = np.sqrt((cell.x - self.target_position[0])**2 +
                                         (cell.y - self.target_position[1])**2)
        if distance_to_target < TARGET_REACHED_THRESHOLD:
            # Reproduce: Create a new cell with the same properties as the current cell
            angle = np.random.uniform(0, 2 * np.pi)
            radius = np.random.uniform(0.01, 0.05)
            new_x = cell.x + radius * np.cos(angle)
            new_y = cell.y + radius * np.sin(angle)
            new_cell = Cell(new_x, new_y, cell.vx, cell.vy, cell.speed)
            self.cells.append(new_cell)

            # Kill a random other cell (if there are more than 10 cells)
            if len(self.cells) > 15:
                random_cell = np.random.choice(self.cells)
                #while random_cell is cell:  # Ensure we don't remove the reproducing cell
                #    random_cell = np.random.choice(self.cells)
                self.cells.remove(random_cell)

            # Set a new target position
            self.target_position = [np.random.uniform(0, 1), np.random.uniform(0, 1)]
            print(f"Target reached! New target position: {self.target_position}")
            return
        
        
# 4. CELL CLASS
class Cell:
    """Represents an individual cell in the simulation."""
    def __init__(self, x, y, vx, vy, speed):
        self.x = x
        self.y = y
        self.vx = vx
        self.vy = vy
        self.ax = 0
        self.ay = 0
        self.speed = speed

    def update_position(self):
        """Update the cell's position based on its velocity."""
        self.x += self.vx
        self.y += self.vy

    def apply_forces(self):
        """Apply a force to the cell, updating its velocity."""
        self.vx += self.ax
        self.vy += self.ay

        # Limit velocity to the maximum allowed
        self.vx = np.clip(self.vx, -MAX_VELOCITY, MAX_VELOCITY)
        self.vy = np.clip(self.vy, -MAX_VELOCITY, MAX_VELOCITY)
        self.ax = 0
        self.ay = 0



            

# 5. Execute the main loop
if __name__ == "__main__":
    main()
